# Noisy Or - an example of intercausal effects
# JMA 4 Aug 2018

EPS = 0.05
STEPS = 10

a = [(x + EPS)/(STEPS + EPS) for x in range(STEPS + 1)]
b = [(x + EPS)/(STEPS + EPS) for x in range(STEPS + 1)]
#print('a: ', ['{:6.4f}'.format(x) for x in a])

The joint for E == True looks like this:

     A      ~A
     --     --
B    ba     b(1-a)

~B  (1-b)a  0

To compute P(B | A E=True) sum the row for B, and normalze by the sum of all terms.

def intercausal(a,b):
    'compute P(B | A E=True) '
    return b / (1 - (1 - b)*(1 - a))

# Generate the matrix for the range of A and B
ic = [[intercausal(x,y) for x in a] for y in b]
nic = np.array(ic)
np.set_printoptions(precision=3, threshold=30)
